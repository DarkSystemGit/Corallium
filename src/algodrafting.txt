struct HeapEntry {
    size: i32,     //+0
    is_free: bool, //+2
    next: i32,     //+3
    prev: i32,     //+5
    data: [i16],   //+7
}
fn alloc(size: i32) -> ptr {
    let ptr = 0;
    while ptr != -1 {
        let csize = readEx(ptr); //size
        let nptr = readEx(ptr + 3); //next
        if csize >= size && (read(ptr + 2) == true) {
            if csize > size + 7 {
                //split block
                let next = ptr + 7 + size; //get block start, current+size+header
                writeEx(next, csize - size - 7); //size
                write(next + 2, true); //mark free
                writeEx(next + 3, readEx(ptr + 3)); //next
                writeEx(next + 5, ptr); //prev
                writeEx(ptr + 3, next); //update current next
                writeEx(ptr, size); //update current size
                if nptr != -1 {
                    writeEx(nptr + 5, next);
                }
            }
            write(ptr + 2, false); //mark used, return ptr
            break;
        }
        if nptr == -1 {
            //alloc new block
            let next = ptr + 7 + csize; //get block start, current+size+header
            writeEx(next, size); //size
            write(next + 2, false); //mark used, return ptr
            writeEx(next + 3, -1); //next
            writeEx(next + 5, ptr); //prev
            writeEx(ptr + 3, next); //update current next
            ptr = next; //return
            break;
        } else {
            ptr = nptr; //continue searching
        }
    }
    return ptr + 7;
}
fn free(ptr: ptr) {
    let ptr = ptr - 7;
    write(ptr + 2, true); //mark free
    let prev = readEx(ptr + 5);
    let next = readEx(ptr + 3);
    if prev != -1 && read(prev + 2) == true {
        //merge with prev
        let nlen = readEx(prev) + readEx(ptr) + 7;
        if next != -1 && read(next + 2) == true {
            //merge with both prev and next
            nlen += readEx(next) + 7;
            next = readEx(next + 3);
        }
        writeEx(prev, nlen); //update prev size to new len
        writeEx(prev + 3, next); //update prev next to new next
        if next != -1 {
            writeEx(next + 5, prev); //update next prev to new prev
        }
    } else if next != -1 && read(next + 2) == true {
        //merge with next only
        let nlen = readEx(ptr) + readEx(next) + 7; //current_size+header_size+next_size
        let nextptr = readEx(next + 3); //new next
        writeEx(ptr, nlen); //update current size to new size including next block
        writeEx(ptr + 3, nextptr); //update current next to new next
        if nextptr != -1 {
            writeEx(nextptr + 5, ptr); //update next prev to this block
        }
    }
}
Register allocation
Given the live ranges at some point:
add v1,v2;live: v1 v2
add v1,v3;live: v3, v1
Build a graph that connects all registers that are live at the same time
     ---v1---
   \/        \/
   v2        v3
v1 -->r1
v2 & v3 -->r2
nodes(virt registers) with no connection can share the same register, nodes with a connection can't.
we can do this by type now
Code			Usage
----			-----
let a:int16=0; 		V1
let b:int32=1;		V1 V2
let d:int32=a+b;	V1 V2 -> V3
let c: int8=5;		Va1(upper 8 bits of V1) V3
let e: int32=d+c	Va1 V3->V4
return e		V4

Graph
-----
V1-V2		r1:V1 r2:V2
V3-Va1	--->	r2:V3 ra1:Va1
  V4		r2:V4
Once we have the graph, we can just color it using Recursive Largest First graph-coloring, then map the colors to registers.
If theirs too many colors, then we spill and try to color again.
Continue until we satisfy the constraints.
